#define CS_ETMVERSION_ETMv4   0x40
 Major version value for ETMv4.
 
#define CS_ETMVERSION(maj, min)   ((maj) | (min))
 
#define CS_ETMVERSION_MAJOR(x)   ((x) & 0xF0)

#define CS_ETMVERSION_MINOR(x)   ((x) & 0x0F)

#define CS_ETMVERSION_IS_ETMV3(x)   ((x) < CS_ETMVERSION_PTM)
 Version is ETMv3.
 
#define CS_ETMVERSION_IS_PTM(x)   (((x) >= CS_ETMVERSION_PTM) && ((x) < CS_ETMVERSION_ETMv4))
 Version is PTM.
